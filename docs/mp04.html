Introduction

With the 2026 Winter Olympics complete and the 2028 Summer Olympics coming to Los Angeles, serious money is being wagered on Team USA’s performance. This report builds a three-factor model to predict how many gold medals Team USA will win in 2028, drawing on historical Olympic data scraped from Olympedia.

Data Acquisition and Preparation

Step 1 — Parse All Olympic Editions

We scrape the editions index at /editions, extracting the edition ID from the year-link href, the olympiad number from the Roman numeral column, the host nation NOC from the flag image URL, and the season (Summer / Winter) from the games-name text.

Code 
parse_editions <- function() {
  page   <- read_olympedia("editions")
  tables <- page |> html_elements("table")

  parse_season_table <- function(tbl, season_label) {
    rows <- tbl |> html_elements("tr")
    bind_rows(lapply(rows, function(row) {
      cells <- row |> html_elements("td")
      if (length(cells) < 4) return(NULL)

      # Cell 1: Roman numeral + href containing edition_id
      cell1_link <- cells[[1]] |> html_elements("a")
      if (length(cell1_link) == 0) return(NULL)

      href           <- cell1_link[[1]] |> html_attr("href")
      edition_id     <- str_extract(href, "\\d+$") |> as.integer()
      olympiad_roman <- cell1_link[[1]] |> html_text(trim = TRUE)

      # Cell 2: Year
      year <- cells[[2]] |> html_text(trim = TRUE) |> as.integer()

      # Cell 4: Host flag (.svg URL)
      flag_img <- cells[[4]] |> html_elements("img")
      host_noc <- if (length(flag_img) > 0) {
        str_extract(flag_img[[1]] |> html_attr("src"), "[A-Z]{3}(?=\\.svg)")
      } else {
        NA_character_
      }

      data.frame(
        edition_id     = edition_id,
        year           = year,
        olympiad_roman = olympiad_roman,
        host_noc       = host_noc,
        season         = season_label,
        stringsAsFactors = FALSE
      )
    }))
  }

  bind_rows(
    parse_season_table(tables[[1]], "Summer"),
    parse_season_table(tables[[2]], "Winter")
  ) |>
    as_tibble() |>
    filter(!is.na(edition_id), !is.na(year)) |>
    mutate(olympiad_num = suppressWarnings(as.integer(as.roman(olympiad_roman))))
}

Step 2 — Sports (Medal Disciplines) per Edition

For each edition we visit /editions/{id} and collect every link matching the pattern /editions/{id}/sports/{CODE}.

Code 
parse_edition_sports <- function(edition_id) {
  empty_tbl <- tibble(
    sport_code = character(),
    sport_name = character(),
    sport_url  = character()
  )

  page <- tryCatch(
    read_olympedia(paste0("editions/", edition_id)),
    error = function(e) NULL
  )
  if (is.null(page)) return(empty_tbl)

  selector    <- sprintf("a[href*='/editions/%d/sports/']", edition_id)
  sport_nodes <- page |> html_elements(selector)

  if (length(sport_nodes) == 0) return(empty_tbl)

  tibble(
    sport_url  = sport_nodes |> html_attr("href"),
    sport_name = sport_nodes |> html_text(trim = TRUE)
  ) |>
    distinct(sport_url, .keep_all = TRUE) |>
    mutate(sport_code = str_extract(sport_url, "[A-Z0-9]+$")) |>
    filter(!is.na(sport_code), sport_code != "") |>
    select(sport_code, sport_name, sport_url)
}

all_sports <- editions |>

Step 3 — Medal Results per Sport × Edition

For each sport page we find the first table containing Olympedia flag images (flag URLs like ...olympedia-flags.../USA.png). We then parse each row as an event record with Gold / Silver / Bronze columns, extracting the three-letter NOC from each flag image src.

Code 
# Pull 3-letter NOC out of the first flag image in a table cell
extract_noc <- function(cell) {
  imgs <- cell |> html_elements("img[src*='olympedia-flags']")
  if (length(imgs) == 0) return(NA_character_)
  str_extract(imgs[[1]] |> html_attr("src"), "[A-Z]{3}(?=\\.(png|svg))")
}

parse_sport_medals <- function(edition_id, sport_code) {
  empty_tbl <- tibble(
    noc    = character(),
    gold   = integer(),
    silver = integer(),
    bronze = integer()
  )

  url  <- paste0("editions/", edition_id, "/sports/", sport_code)
  page <- tryCatch(read_olympedia(url), error = function(e) NULL)
  if (is.null(page) || !inherits(page, "xml_document")) return(empty_tbl)

  # Find the first table that contains flag images (the results table)
  tables      <- page |> html_elements("table")
  results_tbl <- NULL
  for (tbl in tables) {
    if (length(tbl |> html_elements("img[src*='olympedia-flags']")) > 0) {
      results_tbl <- tbl
      break
    }
  }
  if (is.null(results_tbl)) return(empty_tbl)

  # Use tbody rows to skip header automatically
  rows <- results_tbl |> html_elements("tbody tr")
  if (length(rows) == 0) {
    rows <- results_tbl |> html_elements("tr")
    if (length(rows) > 1) rows <- rows[-1]  # drop header
  }
  if (length(rows) == 0) return(empty_tbl)

  raw <- map_dfr(rows, function(row) {
    cells <- row |> html_elements("td")
    if (length(cells) < 2) return(NULL)

    tibble(
      event      = cells[[1]] |> html_text(trim = TRUE),
      gold_noc   = if (length(cells) >= 3) extract_noc(cells[[3]]) else NA_character_,
      silver_noc = if (length(cells) >= 5) extract_noc(cells[[5]]) else NA_character_,
      bronze_noc = if (length(cells) >= 7) extract_noc(cells[[7]]) else NA_character_
    )
  })

  if (nrow(raw) == 0) return(empty_tbl)

  # Pivot → long → aggregate by NOC and medal type
  raw |>
    pivot_longer(
      cols      = ends_with("_noc"),
      names_to  = "medal_type",
      values_to = "noc"
    ) |>
    filter(!is.na(noc), noc != "") |>
    mutate(medal_type = str_remove(medal_type, "_noc")) |>
    count(noc, medal_type, name = "medals") |>
    pivot_wider(
      names_from  = medal_type,
      values_from = medals,
      values_fill = 0L
    )
}

Task 5: Combining Confidence Intervals

We forecast Team USA’s 2028 gold medals using:

\[\text{USA}_{2028} = \underbrace{B}_{\text{baseline}} \times \left(1 + \frac{H}{100}\right) + P \times 22\]

where \(B\) is USA’s average gold count per games (Task 4 Q1), \(H\) is the host nation’s percent increase over the prior same-season Olympics (Task 4 Q2), \(P\) is the probability of winning gold in a debut sport (Task 4 Q3), and 22 is the additional medal events in 2028 vs 2024. We assume the three intervals are independent and combine them via Monte Carlo simulation.

Analysis 2 — Host Nation Effect

For each Summer Olympics we compute the host nation’s gold medal share, then compare it to their average share in the two Games immediately before and after (within ±8 years) to isolate the hosting premium.

Code 
# Gold share for every country at every Summer Olympics
country_share <- olympic_medals |>
  filter(season == "Summer") |>
  group_by(year, noc) |>
  summarise(country_gold = sum(gold), .groups = "drop") |>
  left_join(games_totals |> select(year, total_events), by = "year") |>
  mutate(share = country_gold / total_events)

# Host occasions
host_occasions <- editions |>
  filter(season == "Summer") |>
  select(year, host_noc) |>
  filter(!is.na(host_noc))

# Host bonus = host share minus nearby (non-host) share
compute_host_bonus <- function(noc_code, host_yr) {
  cs <- country_share |> filter(noc == noc_code) |> arrange(year)
  if (nrow(cs) < 2) return(NA_real_)

  host_share <- cs |> filter(year == host_yr) |> pull(share)
  if (length(host_share) == 0) return(NA_real_)

  nearby <- cs |>
    filter(year != host_yr, abs(year - host_yr) <= 8) |>
    pull(share)
  if (length(nearby) == 0) return(NA_real_)

  host_share - mean(nearby)
}

host_bonuses <- host_occasions |>
  mutate(bonus = map2_dbl(host_noc, year, compute_host_bonus)) |>
  filter(!is.na(bonus))

# Summary
host_stats <- host_bonuses |>
  summarise(
    n         = n(),
    mean      = mean(bonus),
    sd        = sd(bonus),
    se        = sd / sqrt(n),
    ci_lo     = mean - qt(0.975, n - 1) * se,
    ci_hi     = mean + qt(0.975, n - 1) * se
  )
