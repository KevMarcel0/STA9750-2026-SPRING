load_state_population <- function(year = 2015:2024) {
    
    # Create folder if it does not exist
    if(!dir.exists(file.path("data", "mp03"))){
        dir.create(file.path("data", "mp03"), showWarnings = FALSE, recursive = TRUE)
    }
    
    # Download and combine data across years
    pop_data <- map_dfr(
        year,
        \(yr) {
            get_acs(
                geography = "state",
                variables = c(total_population = "B01003_001"),
                year = yr,
                geometry = TRUE,
                cache_table = TRUE
            ) |>
                transmute(
                    total_population = estimate,
                    state = NAME,
                    year = yr
                )
        }
    )
    
    pop_data
}

move_row_to_colnames <- function(X, row = 1) {
  X <- as_tibble(X)
  Xrow <- X[row, ]
  X <- X[-row, ]
  colnames(X) <- unlist(Xrow)
