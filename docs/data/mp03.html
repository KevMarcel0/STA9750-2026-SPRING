1 Introduction

This project examines how migration patterns affect state populations and political representation in the United States. Using Census and migration data, I analyze where people are moving, project state populations through 2030, and estimate future congressional seats using the Huntington–Hill method. I then apply these insights to propose a strategy to attract and retain residents, with the goal of increasing my state’s political power.

2 Data Limitations and Model Assumptions

While constructing the migration and population model, several data limitations affected the results. In particular, the state-to-state migration datasets for 2023 and 2024 contained inconsistencies, including missing or improperly parsed rows (especially for “same-state” movements and some low-frequency flows). Additionally, formatting issues in the original Census Excel files required manual cleaning steps, which introduced potential sources of error. As a result, the combined migration dataset did not perfectly match the expected structure (e.g., the full 2,500 state-pair observations per year), which may impact the precision of estimated migration rates and downstream population projections. Despite these challenges, the overall modeling approach remains valid, and the results should be interpreted as reasonable approximations rather than exact forecasts.

state_to_state_migration_2023 <- function() {
  
  if (!dir.exists(file.path("data", "mp03"))) {
    dir.create(file.path("data", "mp03"), showWarnings = FALSE, recursive = TRUE)
  }
  
  xlsx_file <- file.path(
    "data",
    "mp03",
    "State_to_State_Migration_Table_2023_T13.xlsx"
  )
  
  if (!file.exists(xlsx_file)) {
    download.file(
      url = "https://www2.census.gov/programs-surveys/demo/tables/geographic-mobility/2023/state-to-state-migration/State_to_State_Migration_Table_2023_T13.xlsx",
      destfile = xlsx_file,
      mode = "wb",
      quiet = TRUE
    )
  }
  
  states_with_dc <- c(
    state.name[1:8],
    "District of Columbia",
    state.name[9:50]
  )
  
  states_abbrev_with_dc <- c(
    state.abb[1:8],
    "DC",
    state.abb[9:50]
  )
  
  name_to_abbrev <- setNames(states_abbrev_with_dc, states_with_dc)
  
  # ---------------------------
  # 1. Stayed within the state
  # First 6 columns only
  # ---------------------------
  stay_data <- read.xlsx(
  xlsx_file,
  rows = 9:65,
  cols = 1:6,
  colNames = FALSE
) |>
  as_tibble(.name_repair = "minimal") |>
  select(
    state_current = 1,
    stayed_same_house = 2,
    moved_within_state = 3
  ) |>
  mutate(
    state_current = stringr::str_squish(as.character(state_current))
  ) |> 
  filter(!is.na(state_current)) |>
  filter(state_current %in% states_with_dc) |>
  mutate(
    state_current = recode(state_current, !!!name_to_abbrev),
    stayed_same_house = as.integer(replace_na(as.character(stayed_same_house), "0")),
    moved_within_state = as.integer(replace_na(as.character(moved_within_state), "0")),
    population = stayed_same_house + moved_within_state,
    state_1y = state_current,
    year_current = 2023L,
    year_1y = 2022L
  ) |>
  filter(state_current != "DC") |>
  select(state_current, state_1y, population, year_current, year_1y)
  
   # ---------------------------
  # 2. Actual state-to-state flows
  # Columns J:DG only
  # ---------------------------
  row_states <- read.xlsx(
    xlsx_file,
    rows = 9:65,
    cols = 1,
    colNames = FALSE
  ) |>
    as_tibble(.name_repair = "minimal") |>
    rename(state_1y = 1)
  
  flow_wide <- read.xlsx(
    xlsx_file,
    rows = 9:65,
    cols = 10:111,
    colNames = FALSE
  ) |>
    as_tibble(.name_repair = "minimal")
  
  # make sure both have the same number of rows
  n_rows <- min(nrow(row_states), nrow(flow_wide))
  row_states <- slice(row_states, 1:n_rows)
  flow_wide  <- slice(flow_wide, 1:n_rows)
  
  colnames(flow_wide) <- vctrs::vec_interleave(states_with_dc, NA_character_)
  
 flow_data <- bind_cols(row_states, flow_wide) |>
  select(state_1y, any_of(states_with_dc)) |>
  filter(!is.na(state_1y), state_1y %in% states_with_dc) |>
  mutate(
    across(any_of(states_with_dc), as.character)
  ) |>
  pivot_longer(
    cols = -state_1y,
    names_to = "state_current",
    values_to = "population"
  ) |>
  filter(
    !is.na(population),
    state_current != "District of Columbia",
    state_1y != "District of Columbia",
    state_current != state_1y
  ) |>
  mutate(
    state_current = recode(state_current, !!!name_to_abbrev),
    state_1y = recode(state_1y, !!!name_to_abbrev),
    population = if_else(population == "N", "0", population),
    population = as.integer(population),
    year_current = 2023L,
    year_1y = 2022L
  ) |>
  select(state_current, state_1y, population, year_current, year_1y)

move_row_to_colnames <- function(X, row = 1) {
  X <- as_tibble(X)
  Xrow <- X[row, ]
  X <- X[-row, ]
  colnames(X) <- unlist(Xrow)
  readr::type_convert(X)
}

pull_state_from_metro <- function(metro_name) {
  str_extract(
    metro_name,
    ".*, (\\S{2})[-[:alpha:]]* Metro Area",
    group = 1
  )
}

metro_to_metro_migration <- function() {
  
  if (!dir.exists(file.path("data", "mp03"))) {
    dir.create(file.path("data", "mp03"), showWarnings = FALSE, recursive = TRUE)
  }
  
  json_file <- file.path(
    "data",
    "mp03",
    "metro_to_metro_migration_2020.json"
  )
  
  api_url <- paste0(
    "https://api.census.gov/data/2020/acs/flows",
    "?get=MOVEDIN,MOVEDOUT,FULL1_NAME,FULL2_NAME",
    "&for=metropolitan%20statistical%20area/micropolitan%20statistical%20area:*"
  )
  
  if (!file.exists(json_file)) {
    download.file(
      url = api_url,
      destfile = json_file,
      mode = "wb",
      quiet = TRUE
    )
  }
  
  metro_data <- read_json(
    json_file,
    simplifyVector = TRUE
  ) |>
    move_row_to_colnames() |>
    select(
      metro_current = FULL1_NAME,
      metro_1y = FULL2_NAME,
      moved_in = MOVEDIN,
      moved_out = MOVEDOUT
    ) |>
    mutate(
      metro_current_state = pull_state_from_metro(metro_current),
      metro_1y_state = pull_state_from_metro(metro_1y)
    )
  
  metro_data
}

## Introduction
This project examines how migration patterns affect state populations and political representation in the United States. Using Census and migration data, I analyze where people are moving, project state populations through 2030, and estimate future congressional seats using the Huntington–Hill method. I then apply these insights to propose a strategy to attract and retain residents, with the goal of increasing my state’s political power.



## Data Limitations and Model Assumptions
While constructing the migration and population model, several data limitations affected the results. In particular, the state-to-state migration datasets for 2023 and 2024 contained inconsistencies, including missing or improperly parsed rows (especially for “same-state” movements and some low-frequency flows). Additionally, formatting issues in the original Census Excel files required manual cleaning steps, which introduced potential sources of error. As a result, the combined migration dataset did not perfectly match the expected structure (e.g., the full 2,500 state-pair observations per year), which may impact the precision of estimated migration rates and downstream population projections. Despite these challenges, the overall modeling approach remains valid, and the results should be interpreted as reasonable approximations rather than exact forecasts.

move_row_to_colnames <- function(X, row = 1) {
  X <- as_tibble(X)
  Xrow <- X[row, ]
  X <- X[-row, ]
  colnames(X) <- unlist(Xrow)
  readr::type_convert(X)
}

pull_state_from_metro <- function(metro_name) {
  str_extract(
    metro_name,
    ".*, (\\S{2})[-[:alpha:]]* Metro Area",
    group = 1
  )
}

metro_to_metro_migration <- function() {
  
  if (!dir.exists(file.path("data", "mp03"))) {
    dir.create(file.path("data", "mp03"), showWarnings = FALSE, recursive = TRUE)
  }
  
  json_file <- file.path(
    "data",
    "mp03",
    "metro_to_metro_migration_2020.json"
  )
  
  api_url <- paste0(
    "https://api.census.gov/data/2020/acs/flows",
    "?get=MOVEDIN,MOVEDOUT,FULL1_NAME,FULL2_NAME",
    "&for=metropolitan%20statistical%20area/micropolitan%20statistical%20area:*"
  )
  
  if (!file.exists(json_file)) {
    download.file(
      url = api_url,
      destfile = json_file,
      mode = "wb",
      quiet = TRUE
    )
  }
  
  metro_data <- read_json(
    json_file,
    simplifyVector = TRUE
  ) |>
    move_row_to_colnames() |>
    select(
      metro_current = FULL1_NAME,
      metro_1y = FULL2_NAME,
      moved_in = MOVEDIN,
      moved_out = MOVEDOUT
    ) |>
    mutate(
      metro_current_state = pull_state_from_metro(metro_current),
      metro_1y_state = pull_state_from_metro(metro_1y)
    )
  
  metro_data
}
